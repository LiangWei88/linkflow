## 2. 快速开始

### 2.1 使用 Docker Compose（推荐）

#### 步骤 1: 创建工作目录

```bash
mkdir -p ~/linkflow
cd ~/linkflow
```

#### 步骤 2: 创建 docker-compose.yml

```yaml
version: '3.8'

services:
  linkflow:
    image: linkflow/linkflow:latest
    container_name: linkflow
    restart: unless-stopped
    ports:
      - "3000:80"
    volumes:
      - ./data:/app/data
    environment:
      - NODE_ENV=production
      - DATA_DIR=/app/data
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:80/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s
```

#### 步骤 3: 启动服务

```bash
# 拉取并启动
docker-compose up -d

# 查看日志
docker-compose logs -f

# 验证运行状态
docker-compose ps
```

#### 步骤 4: 访问应用

打开浏览器访问: `http://localhost:3000`

---

### 2.2 使用 Docker 命令

```bash
# 创建数据目录
mkdir -p ~/linkflow/data

# 运行容器
docker run -d \
  --name linkflow \
  --restart unless-stopped \
  -p 3000:80 \
  -v ~/linkflow/data:/app/data \
  -e NODE_ENV=production \
  linkflow/linkflow:latest

# 查看日志
docker logs -f linkflow
```

## 3. 生产环境配置

### 3.1 使用 HTTPS（推荐）

#### 方案 A: 使用 Nginx + Let's Encrypt

```yaml
# docker-compose.yml
version: '3.8'

services:
  nginx:
    image: nginx:alpine
    container_name: linkflow-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
      - ./data:/app/data:ro
    depends_on:
      - linkflow

  linkflow:
    image: linkflow/linkflow:latest
    container_name: linkflow
    restart: unless-stopped
    expose:
      - "80"
    volumes:
      - ./data:/app/data
    environment:
      - NODE_ENV=production

  # Let's Encrypt 自动证书
  certbot:
    image: certbot/certbot
    container_name: linkflow-certbot
    volumes:
      - ./certbot/conf:/etc/letsencrypt
      - ./certbot/www:/var/www/certbot
    entrypoint: "/bin/sh -c 'trap exit TERM; while :; do certbot renew; sleep 12h & wait $${!}; done;'"
```

```nginx
# nginx/nginx.conf
server {
    listen 80;
    server_name notes.yourdomain.com;
    
    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }
    
    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2;
    server_name notes.yourdomain.com;
    
    ssl_certificate /etc/letsencrypt/live/notes.yourdomain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/notes.yourdomain.com/privkey.pem;
    
    # SSL 配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;
    
    # Gzip
    gzip on;
    gzip_types text/plain text/css application/json application/javascript;
    
    # 静态文件缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
    
    # 反向代理
    location / {
        proxy_pass http://linkflow:80;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
```

### 3.3 数据备份

#### 自动备份

备份脚本已集成到容器中，默认每日自动备份：

```bash
# 备份位置
./data/backup/notes_2026-03-21.db
./data/backup/notes_2026-03-21.db.gz
```

#### 手动备份

```bash
# 进入容器
docker exec -it linkflow sh

# 执行备份
sqlite3 /app/data/notes.db ".backup '/app/data/backup/manual_$(date +%Y%m%d_%H%M%S).db'"

# 导出为 SQL
sqlite3 /app/data/notes.db .dump > /app/data/backup/export_$(date +%Y%m%d).sql
```

#### 恢复备份

```bash
# 停止服务
docker-compose stop

# 恢复数据
cp ./data/backup/notes_2026-03-21.db ./data/notes.db

# 启动服务
docker-compose start
```

### 4.3 监控与告警

```bash

curl http://localhost:3000/health

# 预期响应
{"status":"ok","version":"1.0.0","timestamp":"2026-03-21T10:00:00Z"}
```

#### 基础监控脚本

```bash
#!/bin/bash
# monitor.sh

HEALTH_URL="http://localhost:3000/health"
WEBHOOK_URL="https://hooks.slack.com/services/YOUR/WEBHOOK/URL"

if ! curl -sf "$HEALTH_URL" > /dev/null; then
    echo "LinkFlow is down!"
    curl -X POST -H 'Content-type: application/json' \
        --data '{"text":"⚠️ LinkFlow 服务异常，请检查！"}' \
        "$WEBHOOK_URL"
    
    # 自动重启
    docker-compose restart
fi
```

添加到 crontab：

```bash
# 每 5 分钟检查一次
*/5 * * * * /path/to/monitor.sh
```

## 5. 故障排查

### 5.1 常见问题

#### 问题 1: 容器无法启动

```bash
# 检查日志
docker-compose logs

# 常见原因：
# 1. 端口被占用
sudo lsof -i :3000

# 2. 权限问题
sudo chown -R $USER:$USER ./data

# 3. 磁盘空间不足
df -h
```

#### 问题 2: 数据库损坏

```bash
# 检查数据库完整性
docker exec -it linkflow sqlite3 /app/data/notes.db "PRAGMA integrity_check;"

# 修复尝试
docker exec -it linkflow sqlite3 /app/data/notes.db ".recover" > /tmp/recover.sql
docker exec -it linkflow sh -c "sqlite3 /app/data/notes.db.fixed < /tmp/recover.sql"
```

#### 问题 3: 性能问题

```bash
# 检查资源使用
docker stats linkflow

# 检查慢查询（进入容器）
docker exec -it linkflow sh
sqlite3 /app/data/notes.db
> .timer on
> SELECT * FROM blocks WHERE doc_id = 'test';
```

## 附录 A: 一键部署脚本

```bash
#!/bin/bash
# install.sh

set -e

INSTALL_DIR="${1:-$HOME/linkflow}"
DOMAIN="${2:-localhost}"
USE_HTTPS="${3:-false}"

echo "🚀 LinkFlow 安装脚本"
echo "安装目录: $INSTALL_DIR"
echo "域名: $DOMAIN"
echo "HTTPS: $USE_HTTPS"
echo ""

# 检查 Docker
if ! command -v docker &> /dev/null; then
    echo "❌ Docker 未安装，请先安装 Docker"
    exit 1
fi

# 创建目录
mkdir -p "$INSTALL_DIR"
cd "$INSTALL_DIR"

# 创建 docker-compose.yml
cat > docker-compose.yml << 'EOF'
version: '3.8'
services:
  linkflow:
    image: linkflow/linkflow:latest
    container_name: linkflow
    restart: unless-stopped
    ports:
      - "3000:80"
    volumes:
      - ./data:/app/data
    environment:
      - NODE_ENV=production
EOF

# 创建数据目录
mkdir -p data/backup

# 启动
echo "📦 启动 LinkFlow..."
docker-compose up -d

# 等待启动
echo "⏳ 等待服务启动..."
sleep 5

# 检查状态
if curl -sf http://localhost:3000/health > /dev/null; then
    echo "✅ LinkFlow 安装成功！"
    echo "访问地址: http://localhost:3000"
else
    echo "❌ 服务启动失败，请检查日志: docker-compose logs"
    exit 1
fi
```

使用方法：

```bash
curl -fsSL https://linkflow.dev/install.sh | bash
# 或
bash install.sh /opt/linkflow notes.example.com true
```
