describe('BlockService', () => {
  let service: BlockService;
  let mockRepo: MockBlockRepository;

  beforeEach(() => {
    mockRepo = new MockBlockRepository();
    service = new BlockService(mockRepo);
  });

  describe('getBlocks', () => {
    it('should return blocks for a given docId', async () => {
      const blocks = await service.getBlocks('2026-03-21');
      expect(blocks).toHaveLength(2);
      expect(mockRepo.findByDoc).toHaveBeenCalledWith('2026-03-21');
    });

    it('should return empty array when doc has no blocks', async () => {
      mockRepo.findByDoc.mockResolvedValue([]);
      const blocks = await service.getBlocks('empty-doc');
      expect(blocks).toEqual([]);
    });

    it('should sort blocks by order', async () => {
      const blocks = await service.getBlocks('doc-1');
      expect(blocks[0].order).toBeLessThan(blocks[1].order);
    });
  });

  describe('createBlock', () => {
    it('should create block with correct structure', async () => {
      const input = { docId: '2026-03-21', content: 'test' };
      const block = await service.createBlock(input);
      
      expect(block.id).toBeDefined();
      expect(block.docId).toBe('2026-03-21');
      expect(block.order).toBe(0);
      expect(block.createdAt).toBeDefined();
    });

    it('should throw ValidationError when docId is empty', async () => {
      await expect(service.createBlock({ docId: '', content: 'test' }))
        .rejects.toThrow(ValidationError);
    });
  });

  describe('moveBlock', () => {
    it('should update parent and order', async () => {
      await service.moveBlock('block-1', 'new-parent', 2);
      
      expect(mockRepo.update).toHaveBeenCalledWith('block-1', {
        parentId: 'new-parent',
        order: 2,
      });
    });

    it('should reorder siblings', async () => {
      await service.moveBlock('block-1', null, 0);
      
      // 验证其他兄弟节点的 order 也被更新
      expect(mockRepo.updateMany).toHaveBeenCalled();
    });
  });
});

// __tests__/integration/database.test.ts
describe('Database Integration', () => {
  let db: Database;

  beforeAll(() => {
    db = new Database(':memory:');
    runMigrations(db);
  });

  afterAll(() => {
    db.close();
  });

  describe('Block CRUD', () => {
    it('should create and retrieve block', () => {
      const id = generateUUID();
      db.prepare('INSERT INTO blocks VALUES (?, ?, ?, ?, ?, ?, ?, ?)')
        .run(id, 'test-doc', 'content', null, 0, 0, Date.now(), Date.now());
      
      const block = db.prepare('SELECT * FROM blocks WHERE id = ?').get(id);
      expect(block.content).toBe('content');
    });

    it('should enforce foreign key constraints', () => {
      expect(() => {
        db.prepare('INSERT INTO backlinks VALUES (?, ?, ?)')
          .run('non-existent', 'also-non-existent', Date.now());
      }).toThrow();
    });

    it('should cascade delete', () => {
      const parentId = generateUUID();
      const childId = generateUUID();
      
      // 创建父子节点
      db.prepare('INSERT INTO blocks VALUES (?, ?, ?, ?, ?, ?, ?, ?)')
        .run(parentId, 'doc', 'parent', null, 0, 0, Date.now(), Date.now());
      db.prepare('INSERT INTO blocks VALUES (?, ?, ?, ?, ?, ?, ?, ?)')
        .run(childId, 'doc', 'child', parentId, 0, 0, Date.now(), Date.now());
      
      // 删除父节点
      db.prepare('DELETE FROM blocks WHERE id = ?').run(parentId);
      
      // 子节点应被级联删除
      const child = db.prepare('SELECT * FROM blocks WHERE id = ?').get(childId);
      expect(child).toBeUndefined();
    });
  });

  describe('FTS5 Search', () => {
    it('should index new blocks', () => {
      db.prepare('INSERT INTO blocks VALUES (?, ?, ?, ?, ?, ?, ?, ?)')
        .run('test-id', 'doc', 'searchable content', null, 0, 0, Date.now(), Date.now());
      
      const results = db.prepare(
        'SELECT * FROM blocks_fts WHERE content MATCH ?'
      ).all('searchable');
      
      expect(results).toHaveLength(1);
    });

    it('should update index on block update', () => {
      db.prepare('UPDATE blocks SET content = ? WHERE id = ?')
        .run('updated content', 'test-id');
      
      const oldResults = db.prepare(
        'SELECT * FROM blocks_fts WHERE content MATCH ?'
      ).all('searchable');
      const newResults = db.prepare(
        'SELECT * FROM blocks_fts WHERE content MATCH ?'
      ).all('updated');
      
      expect(oldResults).toHaveLength(0);
      expect(newResults).toHaveLength(1);
    });
  });
});

### 4.2 Service 集成测试

```typescript
// __tests__/integration/services.test.ts
describe('Service Integration', () => {
  let blockService: BlockService;
  let linkService: LinkService;
  let db: Database;

  beforeEach(() => {
    db = new Database(':memory:');
    runMigrations(db);
    
    const blockRepo = new BlockRepository(db);
    const linkRepo = new LinkRepository(db);
    
    blockService = new BlockService(blockRepo);
    linkService = new LinkService(linkRepo, blockRepo);
  });

  it('should sync backlinks when block is updated', async () => {
    // 创建目标块
    const target = await blockService.createBlock({
      docId: 'doc-1',
      content: 'Target Block',
    });
    
    // 创建源块并引用目标
    const source = await blockService.createBlock({
      docId: 'doc-2',
      content: `Reference to [[${target.id}]]`,
    });
    
    // 同步反链
    await linkService.syncBacklinks(source.id, source.content);
    
    // 验证反链
    const backlinks = await linkService.getBacklinks(target.id);
    expect(backlinks).toHaveLength(1);
    expect(backlinks[0].blockId).toBe(source.id);
  });

  it('should maintain referential integrity', async () => {
    const block = await blockService.createBlock({
      docId: 'doc-1',
      content: 'test',
    });
    
    await linkService.createLink(block.id, 'target-1');
    
    // 删除块应级联删除链接
    await blockService.deleteBlock(block.id);
    
    const backlinks = await linkService.getBacklinks('target-1');
    expect(backlinks).toHaveLength(0);
  });
});
```
