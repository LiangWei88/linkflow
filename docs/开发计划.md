## 2.3 API 接口设计

### 2.3.1 接口规范

所有 API 接口遵循 RESTful 设计规范：

| 方法 | 路径 | 说明 | 请求体 | 响应 |
|------|------|------|--------|------|
| GET | `/api/blocks` | 获取文档的所有块 | - | `Block[]` |
| GET | `/api/blocks/:id` | 获取单个块 | - | `Block` |
| POST | `/api/blocks` | 创建新块 | `{ docId, content, parentId, order }` | `Block` |
| PUT | `/api/blocks/:id` | 更新块内容 | `{ content }` | `Block` |
| DELETE | `/api/blocks/:id` | 删除块 | - | `{ success: boolean }` |
| GET | `/api/backlinks/:targetId` | 获取反链 | - | `Backlink[]` |
| POST | `/api/links` | 创建链接 | `{ blockId, targetId }` | `Link` |
| DELETE | `/api/links/:blockId/:targetId` | 删除链接 | - | `{ success: boolean }` |
| GET | `/api/search` | 搜索内容 | `{ keyword, limit }` | `SearchResult[]` |

### 2.3.2 数据类型

```typescript
// Block 类型
interface Block {
  id: string;
  docId: string;
  content: string;
  parentId: string | null;
  order: number;
  collapsed: boolean;
  createdAt: number;
  updatedAt: number;
}

// Backlink 类型
interface Backlink {
  blockId: string;
  targetId: string;
  createdAt: number;
}

// SearchResult 类型
interface SearchResult {
  block: Block;
  highlights: string[];
  rank: number;
}
```

### 2.3.3 错误处理

所有 API 响应遵循统一的错误格式：

```typescript
interface ApiResponse<T> {
  success: boolean;
  data?: T;
  error?: {
    code: string;
    message: string;
  };
}
```

### 2.3.4 前后端通信流程

```
前端用户操作 (点击编辑 block)
   │
   ▼
React 组件调用 apiClient
   │
   ▼
前端发送 HTTP 请求 (GET /api/blocks 或 PUT /api/blocks/:id)
   │
   ▼
Express 服务端接收请求
   │
   ▼
路由处理 → Service 层 (BlockService)
   │
   ▼
Repository 层 → SQLite (BlockRepository)
   │
   ▼
返回数据 → Express 响应
   │
   ▼
前端接收响应 → 更新 UI (显示编辑器或更新 block 内容)
```

```typescript
// services/LinkService.ts
export class LinkService {
  async syncBacklinks(blockId: string, content: string): Promise<void> {
    const links = this.parseWikilinks(content);
    const existing = await this.getOutgoingLinks(blockId);
    
    // 删除不再存在的链接
    for (const link of existing) {
      if (!links.includes(link)) {
        await this.removeLink(blockId, link);
      }
    }
    
    // 添加新链接
    for (const link of links) {
      if (!existing.includes(link)) {
        await this.createLink(blockId, link);
      }
    }
  }
}
```

```typescript
// services/DailyNoteService.ts
export class DailyNoteService {
  async getOrCreateTodayNote(): Promise<DailyNote> {
    const today = format(new Date(), 'yyyy-MM-dd');
    const existing = await this.blockService.getBlocks(today);
    
    if (existing.length > 0) {
      return { date: today, block: existing[0], isNew: false };
    }
    
    const template = await this.getTemplate('default');
    const content = this.renderTemplate(template, { date: today });
    const block = await this.blockService.createBlock({
      docId: today,
      content,
    });
    
    return { date: today, block, isNew: true };
  }
}
```

```typescript
// services/SearchService.ts
export class SearchService {
  async search(keyword: string, limit = 20): Promise<SearchResult[]> {
    const results = this.db.prepare(`
      SELECT b.*, fts.rank
      FROM blocks b
      JOIN blocks_fts fts ON b.id = fts.rowid
      WHERE fts.content MATCH ?
      ORDER BY fts.rank
      LIMIT ?
    `).all(keyword, limit);
    
    return results.map(r => ({
      block: this.toBlock(r),
      highlights: this.extractHighlights(r.content, keyword),
      rank: r.rank,
    }));
  }
}
```

```typescript
}
```
